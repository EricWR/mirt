#* define prior for latent class analysis
lca_prior <- function(Theta,Etable){
    TP <- nrow(Theta)
    if ( is.null(Etable) ){
    prior <- rep( 1/TP , TP )
    }
    if ( ! is.null(Etable) ){
    prior <- ( rowSums( Etable[ , seq(1,2*I,2)] ) + rowSums( Etable[ , seq(2,2*I,2)]))}
    prior <- prior / sum(prior)
    return(prior)
}

#* estimate model

customTheta=Theta , customPriorFun = lca_prior) ,
